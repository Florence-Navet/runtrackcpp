#include <iostream>

int main() {
    int nombre, inverse = 0;

    std::cout << "Entrez un nombre entier : ";
    std::cin >> nombre;

    int temp = nombre;

    while (temp != 0) {
        int chiffre = temp % 10;          // récupère le dernier chiffre
        inverse = inverse * 10 + chiffre; // ajoute le chiffre inversé
        temp /= 10;                      // enlève le dernier chiffre
    }

    std::cout << "Le nombre inversé de " << nombre << " est " << inverse << std::endl;

    return 0;
}
