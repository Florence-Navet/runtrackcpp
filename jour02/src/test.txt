    #include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int nombre;
    cout << "Entrez un nombre : ";
    cin >> nombre;

    int original = nombre;
    int somme = 0;
    int puissance = 0;

    // Calculer le nombre de chiffres
    while (original != 0)
    {
        original /= 10;
        puissance++;
    }

    original = nombre;

    // Calculer la somme des puissances des chiffres
    while (original != 0)
    {
        int chiffre = original % 10;
        somme += pow(chiffre, puissance);
        original /= 10;
    }

    if (somme == nombre)
        cout << nombre << " est un nombre narcissique." << endl;
    else
        cout << nombre << " n'est pas un nombre narcissique." << endl;

    return 0;
}
